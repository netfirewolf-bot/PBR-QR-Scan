async syncNow() { if (!this.isOnline) { this.notify("🔴 Нет сети", "error"); return; } const btn = document.getElementById('btnSync'); const originalText = btn.innerHTML; btn.disabled = true; btn.innerHTML = '⏳ Отправка...'; this.notify('⏳ Отправка данных на сервер...', 'info'); try { const req = this.db.transaction(['pending'], 'readonly').objectStore('pending').getAll(); req.onsuccess = async () => { const changes = (req.result || []).filter(c => !c.synced); if (changes.length === 0) { this.notify("✅ Нечего отправлять", "success"); btn.disabled = false; btn.innerHTML = originalText; return; } this.log(`Отправка ${changes.length} записей...`, "info"); // ИСПРАВЛЕНИЕ: action теперь в URL, данные в body const response = await fetch(CONFIG.API_URL + '?action=batchUpdate', { method: 'POST', mode: 'cors', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ updates: changes }) }); const result = await response.json(); this.log(`Ответ сервера: ${JSON.stringify(result)}`, "info"); if (result.success && result.updated > 0) { const tx = this.db.transaction(['pending'], 'readwrite'); const store = tx.objectStore('pending'); changes.forEach(c => store.delete(c.id)); tx.oncomplete = () => { this.updatePendingCount(); this.renderPending(); this.playBeep(600, 300); this.notify(`✅ УСПЕШНО! Отправлено: ${result.updated} поз.`, "success"); }; } else { throw new Error(result.error || 'Сервер не подтвердил обновление'); } }; } catch (err) { this.log(`Ошибка синхронизации: ${err.message}`, "error"); this.notify(`❌ Ошибка отправки: ${err.message}`, "error"); } finally { btn.disabled = false; btn.innerHTML = originalText; } }
